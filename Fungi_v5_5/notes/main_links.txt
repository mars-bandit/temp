float SmoothMin( float a, float b, float k ){
	float h = Clamp01( (b-a+k) / (2*k) );
	return a * h + b * (1-h) - k * h * (1-h);
}

// Creates a controllable curve, X = T, bias = 0->1
float BiasFunction( float x, float bias ){
	float k = Pow( 1 - bias, 3 );
	return ( x * k ) / ( x * k - x + 1 );
}
